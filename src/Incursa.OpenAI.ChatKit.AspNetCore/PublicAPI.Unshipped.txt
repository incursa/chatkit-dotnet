#nullable enable
namespace Incursa.OpenAI.ChatKit.AspNetCore
{
    public sealed class ChatKitFontSource
    {
        public ChatKitFontSource() { }
        public string? Display { get; set; }
        public string? Family { get; set; }
        public string? Src { get; set; }
        public string? Style { get; set; }
        public string? UnicodeRange { get; set; }
        public object? Weight { get; set; }
    }

    public sealed class ChatKitHeaderActionOptions
    {
        public ChatKitHeaderActionOptions() { }
        public string? Icon { get; set; }
        public string? OnClickHandler { get; set; }
    }

    public sealed class ChatKitThemeAccentColorOptions
    {
        public ChatKitThemeAccentColorOptions() { }
        public int? Level { get; set; }
        public string? Primary { get; set; }
    }

    public sealed class ChatKitThemeColorOptions
    {
        public ChatKitThemeColorOptions() { }
        public ChatKitThemeAccentColorOptions Accent { get; }
        public ChatKitThemeGrayscaleOptions Grayscale { get; }
        public ChatKitThemeSurfaceColorsOptions Surface { get; }
    }

    public sealed class ChatKitThemeGrayscaleOptions
    {
        public ChatKitThemeGrayscaleOptions() { }
        public int? Hue { get; set; }
        public int? Shade { get; set; }
        public int? Tint { get; set; }
    }

    public sealed class ChatKitThemeOptions
    {
        public ChatKitThemeOptions() { }
        public ChatKitThemeColorOptions Color { get; }
        public ChatKitThemeTypographyOptions Typography { get; }
    }

    public sealed class ChatKitThemeSurfaceColorsOptions
    {
        public ChatKitThemeSurfaceColorsOptions() { }
        public string? Background { get; set; }
        public string? Foreground { get; set; }
    }

    public sealed class ChatKitThemeTypographyOptions
    {
        public ChatKitThemeTypographyOptions() { }
        public int? BaseSize { get; set; }
        public string? FontFamily { get; set; }
        public string? FontFamilyMono { get; set; }
        public System.Collections.Generic.List<Incursa.OpenAI.ChatKit.AspNetCore.ChatKitFontSource> FontSources { get; }
    }

    public sealed class ChatKitHeaderOptions
    {
        public ChatKitHeaderOptions() { }
        public ChatKitHeaderActionOptions? LeftAction { get; set; }
        public ChatKitHeaderActionOptions? RightAction { get; set; }
    }
}

        public int? ThemeBaseSize { get; set; }
        public string? ThemeColorAccentPrimary { get; set; }

        public int? ThemeColorGrayscaleHue { get; set; }
        public int? ThemeColorGrayscaleShade { get; set; }
        public int? ThemeColorGrayscaleTint { get; set; }
        public string? ThemeColorSurfaceBackground { get; set; }
        public string? ThemeColorSurfaceForeground { get; set; }
        public string? ThemeFontFamily { get; set; }
        public string? ThemeFontFamilyMono { get; set; }
        public string? HeaderLeftActionHandler { get; set; }
        public string? HeaderLeftActionIcon { get; set; }
        public string? HeaderRightActionHandler { get; set; }
        public string? HeaderRightActionIcon { get; set; }
    }
}
